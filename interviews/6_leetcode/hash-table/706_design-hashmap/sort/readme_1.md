## üîπ 1. –ë—ã—Å—Ç—Ä–∞—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ (Quicksort)

```c
void quicksort(int arr[], int low, int high) {
    if (low < high) {
        // –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ –∏ –ø–æ–ª—É—á–µ–Ω–∏–µ –∏–Ω–¥–µ–∫—Å–∞ –æ–ø–æ—Ä–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞

        // –†–µ–∫—É—Ä—Å–∏–≤–Ω–∞—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –¥–≤—É—Ö –ø–æ–¥–º–∞—Å—Å–∏–≤–æ–≤

int partition(int arr[], int low, int high) {
    int pivot = arr[high]; // –≤—ã–±–∏—Ä–∞–µ–º –ø–æ—Å–ª–µ–¥–Ω–∏–π —ç–ª–µ–º–µ–Ω—Ç –∫–∞–∫ –æ–ø–æ—Ä–Ω—ã–π
    int i = low - 1; // –∏–Ω–¥–µ–∫—Å –º–µ–Ω—å—à–µ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞

    for (int j = low; j < high; j++) {
        if (arr[j] < pivot) {
            i++;
            // –æ–±–º–µ–Ω arr[i] –∏ arr[j]
            int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp;
        }
    }
    // —Å—Ç–∞–≤–∏–º pivot –Ω–∞ –ø—Ä–∞–≤–∏–ª—å–Ω–æ–µ –º–µ—Å—Ç–æ
    int temp = arr[i + 1]; arr[i + 1] = arr[high]; arr[high] = temp;

    return i + 1; // –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –∏–Ω–¥–µ–∫—Å –æ–ø–æ—Ä–Ω–æ–≥–æ
}
```

## üîπ 2. –°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ —Å–ª–∏—è–Ω–∏–µ–º (Merge Sort)

### üí° –ò–¥–µ—è:

**–†–∞–∑–¥–µ–ª—è–π –∏ –≤–ª–∞—Å—Ç–≤—É–π**: —Ä–∞–∑–±–∏–≤–∞–µ–º –º–∞—Å—Å–∏–≤ –ø–æ–ø–æ–ª–∞–º, —Å–æ—Ä—Ç–∏—Ä—É–µ–º —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–æ –∏ **—Å–ª–∏–≤–∞–µ–º** –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ —á–∞—Å—Ç–∏.

```c
void merge_sort(int arr[], int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;

        // –°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ª–µ–≤–æ–π –∏ –ø—Ä–∞–≤–æ–π —á–∞—Å—Ç–∏

        merge_sort(arr, mid + 1, right);

        // –°–ª–∏—è–Ω–∏–µ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –ø–æ–ª–æ–≤–∏–Ω

void merge(int arr[], int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;

    // –í—Ä–µ–º–µ–Ω–Ω—ã–µ –º–∞—Å—Å–∏–≤—ã
    int L[n1], R[n2];

    // –ö–æ–ø–∏—Ä—É–µ–º –¥–∞–Ω–Ω—ã–µ –≤–æ –≤—Ä–µ–º–µ–Ω–Ω—ã–µ –º–∞—Å—Å–∏–≤—ã
    for (int i = 0; i < n1; i++) L[i] = arr[left + i];
    for (int j = 0; j < n2; j++) R[j] = arr[mid + 1 + j];

    // –°–ª–∏–≤–∞–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–µ –º–∞—Å—Å–∏–≤—ã –æ–±—Ä–∞—Ç–Ω–æ –≤ arr
    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2)
        arr[k++] = (L[i] <= R[j]) ? L[i++] : R[j++];

    // –ö–æ–ø–∏—Ä—É–µ–º –æ—Å—Ç–∞–≤—à–∏–µ—Å—è —ç–ª–µ–º–µ–Ω—Ç—ã
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}
```

## üîπ 3. –°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –∫—É—á–µ–π (Heap Sort)

### üí° –ò–¥–µ—è:

–ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –º–∞—Å—Å–∏–≤ –≤ **max-heap** (–¥–≤–æ–∏—á–Ω—É—é –∫—É—á—É), –∑–∞—Ç–µ–º **—É–¥–∞–ª—è–µ–º –º–∞–∫—Å–∏–º—É–º** (–∫–æ—Ä–µ–Ω—å) –∏ –ø–æ–º–µ—â–∞–µ–º –µ–≥–æ –≤ –∫–æ–Ω–µ—Ü –º–∞—Å—Å–∏–≤–∞.

```c
void heapify(int arr[], int n, int i) {
    int largest = i;        // –∫–æ—Ä–µ–Ω—å
    int left = 2 * i + 1;   // –ª–µ–≤—ã–π —Ä–µ–±—ë–Ω–æ–∫
    int right = 2 * i + 2;  // –ø—Ä–∞–≤—ã–π —Ä–µ–±—ë–Ω–æ–∫

    // –ï—Å–ª–∏ –ª–µ–≤—ã–π –±–æ–ª—å—à–µ –∫–æ—Ä–Ω—è
    if (left < n && arr[left] > arr[largest])
        largest = left;

    // –ï—Å–ª–∏ –ø—Ä–∞–≤—ã–π –±–æ–ª—å—à–µ —Ç–µ–∫—É—â–µ–≥–æ –Ω–∞–∏–±–æ–ª—å—à–µ–≥–æ
    if (right < n && arr[right] > arr[largest])
        largest = right;

    // –ï—Å–ª–∏ –Ω–∞–∏–±–æ–ª—å—à–∏–π –Ω–µ –∫–æ—Ä–µ–Ω—å
    if (largest != i) {
        int temp = arr[i]; arr[i] = arr[largest]; arr[largest] = temp;

        // –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ heapify –ø–æ–¥–¥–µ—Ä–µ–≤–æ
        heapify(arr, n, largest);
    }
}

void heap_sort(int arr[], int n) {
    // –°—Ç—Ä–æ–∏–º max-heap
    for (int i = n / 2 - 1; i >= 0; i--)
        heapify(arr, n, i);

    // –ò–∑–≤–ª–µ–∫–∞–µ–º —ç–ª–µ–º–µ–Ω—Ç—ã –∏–∑ –∫—É—á–∏ –ø–æ –æ–¥–Ω–æ–º—É
    for (int i = n - 1; i > 0; i--) {
        // –ü–µ—Ä–µ–º–µ—â–∞–µ–º —Ç–µ–∫—É—â–∏–π –∫–æ—Ä–µ–Ω—å –≤ –∫–æ–Ω–µ—Ü

        // –≤—ã–∑—ã–≤–∞–µ–º heapify –Ω–∞ —É–º–µ–Ω—å—à–µ–Ω–Ω–æ–π –∫—É—á–µ
        heapify(arr, i, 0);
    }
}
```

### üß† –ö–∞–∫ —Ä–∞–±–æ—Ç–∞–µ—Ç:
