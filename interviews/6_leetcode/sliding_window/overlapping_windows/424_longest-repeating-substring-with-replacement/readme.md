    count := [26]int{}
    left, maxCount, maxLength := 0, 0, 0

    for right := 0; right < len(s); right++ {
        count[s[right]-'A']++
        if count[s[right]-'A'] > maxCount {
            maxCount = count[s[right]-'A']
        }

        }

        }
    }

}
